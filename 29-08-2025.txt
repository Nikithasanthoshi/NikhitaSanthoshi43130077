        WHEN Marks < 25 THEN 'F'

        WHEN Marks BETWEEN 45 AND 49 THEN 'D'
        WHEN Marks BETWEEN 50 AND 59 THEN 'C'
        WHEN Marks BETWEEN 60 AND 80 THEN 'B'

    END AS Grade

select 
    upper(name) as Student_Name,
    age,
    skill,
    Marks,
    case
        when Marks < 25 then 'F'
        when Marks between 25 and 44 then 'E'
        when Marks between 45 and 49 then 'D'
        when Marks between 50 and 59 then 'C'
        when Marks between 60 and 80 then 'B'
        else 'A'
    end as Grade
from ECE;
select 
    lower(skill) as Student_Skill,
    age,
    name,
    Marks,
    case
        when Marks < 25 then 'F'
        when Marks between 25 and 44 then 'E'
        when Marks between 45 and 49 then 'D'
        when Marks between 50 and 59 then 'C'
        when Marks between 60 and 80 then 'B'
        else 'A'
    end as Grade
from ECE;
SELECT 
    e.name AS Ece_Student,
    e.skill AS Ece_Skill,
    e.marks AS Ece_Marks,
    c.name AS Cse_Student,
    c.skill AS Cse_Skill,
    c.marks AS Cse_Marks
FROM Ece e
INNER JOIN Cse c
ON e.skill = c.skill;
SELECT 
    e.name AS Ece_Student,
    e.skill AS Ece_Skill,
    c.name AS Cse_Student,
    c.skill AS Cse_Skill
FROM Ece e
LEFT JOIN Cse c
ON e.skill = c.skill;
SELECT 
    c.name AS Cse_Student,
    c.skill AS Cse_Skill,
    e.name AS Ece_Student,
    e.skill AS Ece_Skill
FROM Ece e
RIGHT JOIN Cse c
ON e.skill = c.skill;
SELECT 
    e.name AS Ece_Student,
    e.skill AS Ece_Skill,
    c.name AS Cse_Student,
    c.skill AS Cse_Skill
FROM Ece e
LEFT JOIN Cse c
ON e.skill = c.skill
UNION
SELECT 
    e.name AS Ece_Student,
    e.skill AS Ece_Skill,
    c.name AS Cse_Student,
    c.skill AS Cse_Skill
FROM Ece e
RIGHT JOIN Cse c
ON e.skill = c.skill;
